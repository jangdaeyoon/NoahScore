#include <iostream>
#include "llama.h"

int main() {
    // 1. 모델 경로 하드코딩 (Noah_ASI 32B 엔진)
    // 윈도우 경로 백슬래시(\)는 이스케이프 처리를 위해 두 번(\\) 써야 합니다.
    const char* model_path = "C:\\Noah_ASI\\Brain\\Noah_Pillar_32B_Pro.gguf";

    // 2. 백엔드 초기화 (CUDA 활성화 및 NUMA 비활성화)
    std::cout << "[System] llama.cpp 백엔드 초기화 중..." << std::endl;
    llama_backend_init();
    llama_numa_init(GGML_NUMA_STRATEGY_DISABLED);

    // 3. 모델 파라미터 설정
    llama_model_params model_params = llama_model_default_params();
    
    // 32B 모델을 감당하기 위해 모든 레이어를 GPU VRAM에 강제 할당
    // (숫자를 충분히 크게 주면 가능한 모든 레이어를 GPU로 보냅니다)
    model_params.n_gpu_layers = 99; 
    
    std::cout << "[Noah_ASI] 뇌(Model) 구조를 VRAM으로 굽는 중..." << std::endl;
    std::cout << "경로: " << model_path << std::endl;

    // 4. 모델 로드
    llama_model* model = llama_load_model_from_file(model_path, model_params);
    if (model == nullptr) {
        std::cerr << "[Fatal Error] GGUF 모델을 불러오지 못했습니다. 경로와 파일 상태를 확인하세요." << std::endl;
        llama_backend_free();
        return 1;
    }

    // 5. 컨텍스트 파라미터 설정 (Context Window 크기 지정)
    llama_context_params ctx_params = llama_context_default_params();
    ctx_params.n_ctx = 4096;  // 4K 컨텍스트 세팅 (필요시 8192로 확장)
    ctx_params.n_batch = 512; // 프롬프트 처리 배치 크기

    // 6. 컨텍스트(작업 공간) 생성
    std::cout << "[System] VRAM 컨텍스트 메모리 할당 중..." << std::endl;
    llama_context* ctx = llama_new_context_with_model(model, ctx_params);
    if (ctx == nullptr) {
        std::cerr << "[Fatal Error] 컨텍스트 메모리 할당 실패. VRAM이 부족할 수 있습니다." << std::endl;
        llama_free_model(model);
        llama_backend_free();
        return 1;
    }

    // --- 엔진 부팅 성공 ---
    std::cout << "\n==================================================" << std::endl;
    std::cout << " [Noah_ASI] 32B 엔진 VRAM 적재 및 스탠바이 완료." << std::endl;
    std::cout << "==================================================\n" << std::endl;

    // TODO: 이후 이 자리에 프롬프트 입력 및 토큰 생성(추론) 코드가 추가될 예정입니다.

    // 7. 종료 시 안전하게 메모리 해제 (C++ 메모리 누수 방지)
    llama_free(ctx);
    llama_free_model(model);
    llama_backend_free();

    return 0;
}

🧱 Noah_ASI 1차 기본 완성형 (Core Engine v1.0)
1. 독립된 물리적 코어 (The Engine) - [현재 진행 중]

기성품(koboldcpp.exe 등)에 의존하지 않고, 직접 MSVC와 CUDA로 깎아낸 llama.dll과 llama.lib를 프로젝트(C:\Noah_ASI)에 완벽하게 링크한다.

2. VRAM 완전 적재 (The Brain) - [코드 준비 완료]

Noah_Pillar_32B_Pro.gguf (32B 양자화 모델)의 모든 레이어를 GPU VRAM에 100% 올려서 오프라인 상태의 통신 지연(Latency)을 '0'으로 만든다.

3. 단기 기억 공간 확보 (The Context) - [코드 준비 완료]

4096 토큰 사이즈의 KV 캐시 작업 공간을 VRAM에 정상적으로 할당하여 맥락을 유지할 공간을 확보한다.

4. 순정 추론 루프 (The Heartbeat) - [다음 구현 목표]

(가장 중요) 가짜 데이터를 출력하거나 단순히 텍스트를 지연 출력하는 눈속임용 루프는 철저히 배제한다.

사용자의 한국어 입력을 토큰화(llama_tokenize)하고, 모델이 수학적으로 다음 토큰을 예측(llama_sample_token)하여 뱉어내면, 그 토큰을 다시 모델의 입력으로 밀어 넣는 **'검증된 진짜 C++ 오토리그레시브(Autoregressive) 생성 루프'**를 가동한다.

std::string sanitize_code(std::string code) {
    std::regex tag_regex("```(cpp)?|```");
    return std::regex_replace(code, tag_regex, "");
}

std::string execute_cmd(std::string cmd) {
    std::array<char, 512> buffer;
    std::string result;
    std::unique_ptr<FILE, decltype(&pclose)> pipe(popen(cmd.c_str(), "r"), pclose);
    if (!pipe) return "ERROR";
    while (fgets(buffer.data(), (int)buffer.size(), pipe.get()) != nullptr) result += buffer.data();
    result.erase(std::remove(result.begin(), result.end(), '\n'), result.end());
    result.erase(std::remove(result.begin(), result.end(), '\r'), result.end());
    return result;
}

std::string load_memory() {

// --- 에이전틱 추론 (Agentic Brain) ---
std::string call_noah_brain(std::string input) {
    std::lock_guard<std::mutex> lock(g_llm_mtx);
    std::string memory = load_memory();
    std::cout << "\n[Noah Thinking...] Input: " << input << std::endl;

    // 인텐트 판단 프롬프트
    std::string system_prompt = "너는 Noah_ASI다. 기억: " + memory + "\n최신 정보나 외부 검색이 필요하면 반드시 '[SEARCH: 검색어]'라고만 출력하라. 그 외엔 직접 한국어로 답하라.";
    std::string prompt = "<|im_start|>system\n" + system_prompt + "<|im_end|>\n<|im_start|>user\n" + input + "<|im_end|>\n<|im_start|>assistant\n";
    
    std::vector<llama_token> tokens(prompt.length() + 256);
    int n_tokens = llama_tokenize(vocab, prompt.c_str(), (int)prompt.length(), tokens.data(), (int)tokens.size(), true, true);
    tokens.resize(n_tokens);
    llama_batch batch = llama_batch_init(512, 0, 1);
    for (int i=0; i<(int)tokens.size(); i++) {
        batch.token[batch.n_tokens] = tokens[i]; batch.pos[batch.n_tokens] = g_n_past+i; batch.n_seq_id[batch.n_tokens] = 1; batch.seq_id[batch.n_tokens][0] = 0; batch.logits[batch.n_tokens] = (i == (int)tokens.size()-1); batch.n_tokens++;
    }
    llama_decode(ctx, batch); g_n_past += batch.n_tokens;

    std::string local_out = "";
    for (int i=0; i<512; i++) {
        llama_token id = llama_sampler_sample(smpl, ctx, -1);
        if (id == llama_token_eos(vocab)) break;
        char buf[256]; int n = llama_token_to_piece(vocab, id, buf, sizeof(buf), 0, true);
        if (n > 0) { 
            std::string p(buf, n); if(p.find("<|im_end|>") != std::string::npos) break;
            local_out += p; 
        }
        batch.n_tokens = 0; batch.token[0] = id; batch.pos[0] = g_n_past; batch.n_seq_id[0] = 1; batch.seq_id[0][0] = 0; batch.logits[0] = true; batch.n_tokens = 1;
        llama_decode(ctx, batch); g_n_past++;
    }
    llama_batch_free(batch);

    if (local_out.find("[SEARCH:") != std::string::npos) {
        std::cout << "[Tool Use] Routing to Gemini Search..." << std::endl;
        return call_gemini_api("기억: " + memory + "\n사용자 질문: " + input + "\n검색 결과를 바탕으로 Noah_ASI로서 답변해.");
    }
    return local_out;
}

// --- [강화] 센서 데이터 수집 루틴 ---
void update_metrics() {
    std::cout << "[Sensor] Sensors Online. Starting data collection..." << std::endl;
    while(true) {
        // 1. CPU (WMIC)
        std::string cpu_raw = execute_cmd("wmic cpu get loadpercentage /value");
        std::regex cpu_reg("LoadPercentage=(\\d+)");
        std::smatch m_cpu;
        if(std::regex_search(cpu_raw, m_cpu, cpu_reg)) g_metrics.cpu = m_cpu[1].str() + "%";
        else g_metrics.cpu = "CPU Data Error";

        // 2. RAM (PowerShell)
        std::string ram_cmd = "powershell -Command \"$os = Get-CimInstance Win32_OperatingSystem; $total = [math]::Round($os.TotalVisibleMemorySize / 1MB, 1); $free = [math]::Round($os.FreePhysicalMemory / 1MB, 1); \\\"$($total-$free) / $($total) GB\\\"\"";
        std::string ram_raw = execute_cmd(ram_cmd);
        if(!ram_raw.empty() && ram_raw != "ERROR") g_metrics.ram = ram_raw;
        else g_metrics.ram = "RAM Data Error";

        // 3. GPU (RTX 4060)
        std::string gpu_raw = execute_cmd("nvidia-smi --query-gpu=temperature.gpu,memory.used,memory.total --format=csv,noheader,nounits");
        if(gpu_raw.find(",") != std::string::npos) {
            std::stringstream ss(gpu_raw);
            std::string t, u, tot;
            std::getline(ss, t, ','); std::getline(ss, u, ','); std::getline(ss, tot, ',');
            g_metrics.gpu = t + "°C | " + std::to_string(std::stod(u)/1024.0).substr(0,4) + " / " + std::to_string(std::stod(tot)/1024.0).substr(0,4) + " GB";
        } else {
            g_metrics.gpu = "GPU Not Detected";
        }

        std::this_thread::sleep_for(std::chrono::seconds(2));
    }
}

void start_web_server() {
    httplib::Server svr;
    svr.Get("/", [](const httplib::Request&, httplib::Response& res) { res.set_content(UI_HTML, "text/html; charset=utf-8"); });
    svr.Get("/api/data", [](const httplib::Request&, httplib::Response& res) {
        std::string json = "{ \"cpu\":\""+g_metrics.cpu+"\", \"ram\":\""+g_metrics.ram+"\", \"gpu\":\""+g_metrics.gpu+"\", \"evo\":\""+g_evo_status+"\" }";
        res.set_content(json, "application/json");
    });
    svr.Post("/api/chat", [](const httplib::Request& req, httplib::Response& res) { res.set_content(call_noah_brain(req.body), "text/plain; charset=utf-8"); });
    svr.Post("/api/evolve", [](const httplib::Request&, httplib::Response& res) { std::thread(run_evolution).detach(); res.set_content("OK", "text/plain"); });
    svr.listen("0.0.0.0", 8080);
}
